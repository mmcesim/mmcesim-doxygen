148// random string
149// Reference: https://stackoverflow.com/a/24586587/15080514
150static std::string randomString(std::string::size_type length) {
151 static auto& chrs = "0123456789"
152 "abcdefghijklmnopqrstuvwxyz"
153 "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
154 thread_local static std::mt19937 rg{ std::random_device{}() };
155 thread_local static std::uniform_int_distribution<std::string::size_type> pick(0, sizeof(chrs) - 2);
156 std::string s;
157 s.reserve(length);
158 while (length--) { s += chrs[pick(rg)]; }
159 return s;
160}
