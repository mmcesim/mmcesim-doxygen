208template <typename T>
209inline T Export::_as(const YAML::Node& n, bool mattered) {
210 bool l;
211 if (std::is_same_v<T, int>) {
212 l = _preCheck(n, DType::INT, mattered);
213 } else if (std::is_same_v<T, double>) {
214 l = _preCheck(n, DType::DOUBLE, mattered);
215 } else if (std::is_same_v<T, std::string>) {
216 l = _preCheck(n, DType::STRING, mattered);
217 } else if (std::is_same_v<T, bool>) {
218 l = _preCheck(n, DType::BOOL, mattered);
219 } else if (std::is_same_v<T, char>) {
220 l = _preCheck(n, DType::CHAR, mattered);
221 }
222 if (l) return n.as<T>();
223 else if (mattered) throw("Invalid!");
224 else return T();
225}
