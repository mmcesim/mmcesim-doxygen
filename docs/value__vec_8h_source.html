72template <typename T>
73void Value_Vec<T>::parseToken(const std::string& s) {
74 // find first ':'
75 size_t first_colon = s.find(':');
76 size_t second_colon = std::string::npos;
77 if (first_colon != std::string::npos) {
78 second_colon = s.find(':', first_colon + 1);
79 if (second_colon != std::string::npos) {
80 // 2 colons
81 T v1 = strAs<T>(s.substr(0, first_colon));
82 T v_ = strAs<T>(s.substr(first_colon + 1, second_colon - first_colon));
83 T v2 = strAs<T>(s.substr(second_colon + 1, s.size() - 1 - second_colon));
84 if (v_ == 0) {
85 if (v1 == v2) {
86 _data.push_back(strAs<T>(s));
87 } else {
88 // ERROR! step is stuck at 0.
89 // TODO: error handling
90 assert("step size as 0 in 2 colons specification.");
91 }
92 } else if (v_ > 0) {
93 for (T v = v1; v <= v2 + 1E-12; v += v_) { _data.push_back(v); }
94 } else {
95 for (T v = v1; v + 1E-12 >= v2; v += v_) { _data.push_back(v); }
96 }
97 } else {
98 // 1 colon
99 T v1 = strAs<T>(s.substr(0, first_colon));
100 T v2 = strAs<T>(s.substr(first_colon + 1, s.size() - 1 - first_colon));
101 if (v1 <= v2) {
102 for (T v = v1; v <= v2 + 1E-12; v += 1) { _data.push_back(v); }
103 } else {
104 // avoid the unsigned problem when subtracting from 0;
105 for (T v = v2 + 1; v + 1E-12 >= v1 + 1; v -= 1) { _data.push_back(v - 1); }
106 }
107 }
108 } else {
109 _data.push_back(strAs<T>(s));
110 }
111}
